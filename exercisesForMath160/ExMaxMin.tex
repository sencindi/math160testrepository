%%MinTF1
\begin{exercise}
	If $f'(a) = 0$, then $x=a$ is a local extremum of $f$.
	\begin{multipleChoice}	
		\choice{True}
		\choice[correct]{False}
	\end{multipleChoice}
\end{exercise}


%%MinTF4
\begin{exercise}
If $f$ is continuous, $f'(x)<0$ when $x<a$ and $f'(x) \geq 0$ when $x>a$, then $a$ is a local max.
	\begin{multipleChoice}	
		\choice{True}
		\choice[correct]{False}
	\end{multipleChoice}
\end{exercise}

%%MinTF3
\begin{exercise}
A function on a \textbf{closed} interval must have a local extremum.
	\begin{multipleChoice}	
		\choice{True}
		\choice[correct]{False}
	\end{multipleChoice}
\end{exercise}

%%MinTF5
\begin{exercise}
If $f$ has a local max at $x=a$, then its square $f^2$ must also have a local max at $x=a$
	\begin{multipleChoice}	
		\choice{True}
		\choice[correct]{False}
	\end{multipleChoice}

\end{exercise}

%%MinTF6
\begin{exercise}
	If $f'(a)$ does not exist, then $f$ cannot have a local extremum at $x=a$

	\begin{multipleChoice}	
		\choice{True}
		\choice[correct]{False}
	\end{multipleChoice}
\end{exercise}

%%MinTF7
\begin{exercise}
If $f$ and $g$ both have local minima at $x=a$, then their sum $f+g$ has a local minima at $x=a$ as well.
	\begin{multipleChoice}	
		\choice[correct]{True}
		\choice{False}
	\end{multipleChoice}
\end{exercise}

%%MinTF8
\begin{exercise}
If $f''(x)<0$ on $(2,5)$, and $f'(3)=0$, then $x=3$ is a local max
	\begin{multipleChoice}	
		\choice[true]{True}
		\choice{False}
	\end{multipleChoice}
\end{exercise}

%%MinTF9
\begin{exercise}
If $f$ is differentiable and decreasing on $(a,b)$, then $f'(x)<0$ on $(a,b)$

	\begin{multipleChoice}	
		\choice[correct]{True}
		\choice{False}
	\end{multipleChoice}
\end{exercise}

%%MinTF10
\begin{exercise}
If $f$ and $g$ both have local minima at $x=a$, then their product $f\cdot g$ has a local minima at $x=a$ as well.

\begin{multipleChoice}	
  \choice{True}
  \choice[correct]{False}
\end{multipleChoice}
\end{exercise}
